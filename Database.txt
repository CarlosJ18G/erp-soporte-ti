CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

CREATE TYPE user_role AS ENUM ('admin', 'coordinador', 'tecnico', 'cliente');
CREATE TYPE ticket_status AS ENUM ('abierto', 'en_proceso', 'cerrado', 'cancelado');
CREATE TYPE ticket_priority AS ENUM ('baja', 'media', 'alta', 'critica');
CREATE TYPE asset_status AS ENUM ('operativo', 'en_mantenimiento', 'fuera_de_servicio', 'obsoleto');
CREATE TYPE service_order_status AS ENUM ('pendiente', 'asignada', 'en_proceso', 'finalizada', 'cancelada');

CREATE TABLE clients (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    name VARCHAR(150) NOT NULL,
    tax_id VARCHAR(50),
    email VARCHAR(150),
    phone VARCHAR(50),
    address TEXT,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    deleted_at TIMESTAMP NULL
);

CREATE TABLE technicians (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    first_name VARCHAR(100) NOT NULL,
    last_name VARCHAR(100) NOT NULL,
    email VARCHAR(150) UNIQUE NOT NULL,
    phone VARCHAR(50),
    specialty VARCHAR(150),
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    deleted_at TIMESTAMP NULL
);

CREATE TABLE assets (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    client_id UUID NOT NULL,
    asset_type VARCHAR(100) NOT NULL,
    brand VARCHAR(100),
    model VARCHAR(100),
    serial_number VARCHAR(100),
    location VARCHAR(150),
    status asset_status DEFAULT 'operativo',
    installation_date DATE,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    deleted_at TIMESTAMP NULL,
    CONSTRAINT fk_assets_client
        FOREIGN KEY (client_id)
        REFERENCES clients(id)
        ON DELETE CASCADE
);

CREATE TABLE tickets (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    client_id UUID NOT NULL,
    asset_id UUID,
    title VARCHAR(200) NOT NULL,
    description TEXT,
    status ticket_status DEFAULT 'abierto',
    priority ticket_priority DEFAULT 'media',
    opened_at TIMESTAMP DEFAULT NOW(),
    closed_at TIMESTAMP NULL,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    deleted_at TIMESTAMP NULL,
    CONSTRAINT fk_ticket_client
        FOREIGN KEY (client_id)
        REFERENCES clients(id)
        ON DELETE CASCADE,
    CONSTRAINT fk_ticket_asset
        FOREIGN KEY (asset_id)
        REFERENCES assets(id)
        ON DELETE SET NULL
);

CREATE TABLE service_orders (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    ticket_id UUID NOT NULL,
    technician_id UUID,
    status service_order_status DEFAULT 'pendiente',
    assigned_at TIMESTAMP,
    started_at TIMESTAMP,
    completed_at TIMESTAMP,
    notes TEXT,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    deleted_at TIMESTAMP NULL,
    CONSTRAINT fk_order_ticket
        FOREIGN KEY (ticket_id)
        REFERENCES tickets(id)
        ON DELETE CASCADE,
    CONSTRAINT fk_order_technician
        FOREIGN KEY (technician_id)
        REFERENCES technicians(id)
        ON DELETE SET NULL
);

CREATE TABLE work_logs (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    service_order_id UUID NOT NULL,
    technician_id UUID NOT NULL,
    hours_worked NUMERIC(5,2) NOT NULL,
    work_description TEXT,
    work_date DATE DEFAULT CURRENT_DATE,
    created_at TIMESTAMP DEFAULT NOW(),
    CONSTRAINT fk_worklog_order
        FOREIGN KEY (service_order_id)
        REFERENCES service_orders(id)
        ON DELETE CASCADE,
    CONSTRAINT fk_worklog_technician
        FOREIGN KEY (technician_id)
        REFERENCES technicians(id)
        ON DELETE CASCADE
);

CREATE TABLE materials_used (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    service_order_id UUID NOT NULL,
    material_name VARCHAR(150) NOT NULL,
    quantity INTEGER NOT NULL,
    unit_cost NUMERIC(10,2),
    created_at TIMESTAMP DEFAULT NOW(),
    CONSTRAINT fk_material_order
        FOREIGN KEY (service_order_id)
        REFERENCES service_orders(id)
        ON DELETE CASCADE
);

CREATE INDEX idx_assets_client_id ON assets(client_id);
CREATE INDEX idx_tickets_client_id ON tickets(client_id);
CREATE INDEX idx_tickets_asset_id ON tickets(asset_id);
CREATE INDEX idx_orders_ticket_id ON service_orders(ticket_id);
CREATE INDEX idx_orders_technician_id ON service_orders(technician_id);
